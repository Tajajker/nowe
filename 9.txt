 	

#include <stdio.h>
#include <stdlib.h>
#define ROZ 20

int palindrom(char* tab ,int n )
{
if(n==1)
    {
    return 1;
    }
else if(tab[0]!=tab[n-1])
   {
    return 0;
   }
return palindrom((tab+1),(n-2));
}
int main()
{
int n=0;
char tab[ROZ];
printf("Wpisz ci¹g znaków.\n");
fgets(tab,ROZ,stdin);
for(int i=0;tab[i]!='\0' && tab[i]!='\n';i++)
   {
    n=n+1;
   }
if (n==0)
   {
    printf("Nie podano ci¹gu znaków.\n");
   }
if( palindrom(tab , n )== 0)
   {
    printf("Ci¹g znaków nie jest palindromem");
   }
else
   {
    printf("Ci¹g znaków jest palindromem");
   }
}
